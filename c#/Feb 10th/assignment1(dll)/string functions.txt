using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Myfunctions
{
    public class Stringfunctions
    {
        public string CombineStrings(string string1, string string2)
        {
            return string1 + " " + string2;
        }

        //1. Find no.of characters in a sentence
        public int StringLength(string string1)
        {
            return string1.Length;

        }

        // 2. Check for Palindrome
        public bool Palindrome(string string2)
        {
            string revs="";
            for (int i = string2.Length - 1; i >= 0; i--)   
            {
                revs += string2[i].ToString();
            }
            if(revs==string2)
            {
                return true;
            }
            else
                return false;
        }

        //3.Reverse a sentence
        public string Reversesentence(string sentence,string reverse= "")
        {
            for(int i=sentence.Length-1;i>=0;i--)
            {
                reverse = reverse + sentence[i].ToString();

            }
            return reverse;
        }

        //4. Find no.of vowels, consonants, integers and sp characters in a sentence

        public string CountCharacters(string string3)
        {
            int CountVowels = 0;
            int CountConsonants = 0;
            int CountIntegers = 0;
            int CountSpecial = 0;
            foreach(char c in string3)
            {
                if(!char.IsWhiteSpace(c))
                {
                    if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' || c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U')
                        CountVowels++;
    
                    else if ((c >= 'a' && c<='z') || (c>= 'A' && c<= 'Z'))
                    {
                        CountConsonants++;
                    }
                    else if (c >= '0' && c <= '9')
                    {
                        CountIntegers++;
                   }
                    else
                    {
                        CountSpecial++;
                    }
                }

            }
            return "no. of vowels= " + CountVowels + "\nno. of consonants= " + CountConsonants + "\nno. of integers= " +
                   CountIntegers + "\nno. of special= " + CountSpecial;
        }


        // 5. Change case to upper case
        public string ToUpperCase(string str)
        {
            return str.ToUpper();
        }

        //6.Change sentence to proper case
        public string ToProperCase(string str)
        {
            string proper = "";
            proper += (char.ToUpper(str[0]));

            for (int i = 1; i < str.Length; i++)
            {
                if (char.IsWhiteSpace(str[i - 1]))
                {
                    proper += char.ToUpper(str[i]);
                }
                else
                    proper += str[i];
            }
            return proper;
        }

        //7.Combine two sentences
        public string CombineSentences(string sentence1, string sentence2)
        {
            return sentence1 + " " + sentence2;
        }

        // 9. Count no.of words in a sentence
        public int noofwords(string str)
        {
            int count = 0;
            foreach(char c in str)
            {
                if (char.IsWhiteSpace(c) || c == '\n' || c == '\t')
                    count++;
            }
            return count;
        }

        //10.Search a substring in a sentence
        public string Substring(string str, string substring)
        {
            if (str.Contains(substring) == true)
                return "The given substring " +  substring + " is present in the string";
            else
                return "The given substring" +  substring + " is not present in the string";
        }

        //11.Find the no. of occurences of a substring in a sentence
        public int FindOccurence(string str, string substring)
        {
            int count = 0;
            int a = 0;

            while ((a = str.IndexOf(substring, a)) != -1)
            {
                a += substring.Length;
                count++;
            }
            return count;
        }

    }
}
